%Enable optional arguments, for the experienceitems below
\newcommand{\optarg}[2][]{%
  \ifthenelse{\isempty{#1}}%
    {}% if #1 is empty
    {#2}% if #1 is not empty
}

%Main section - Qualifications, Education, Experience.

\newenvironment{rsection}[1]{%
  \noindent
  {\Large \textbf{#1}}\\
  \rule[8pt]{\columnwidth}{0.5pt}
}{
}

%Subsection within main section - Skills, Projects, etc
%Encloses several \items in an itemize list
\newenvironment{rsubsection}[1]{%
  \noindent
  {\large \textbf{#1}}
  \raggedright{}
}{
  \vspace{0.5em}
}

\newenvironment{rsubitems}[2][]{%
  \begin{rsubsection}{#2}
  \begin{itemize}[#1]
}{
  \end{itemize}
  \end{rsubsection}
}

\newenvironment{rpara}[1]{%
  \noindent
  \optarg[#1]{\large{\textbf{#1}}}
  \normalsize
  \begin{sloppypar}%
  \noindent%
}{
  \end{sloppypar}
  \vspace{1em}
}

\newcommand{\rexperiencehead}[5]{%
  \large \textbf{#1}%
  \hfill
  \normalsize
  \optarg[#4]{#4}%
  \optarg[#5]{\ -- #5}% chktex 8

  \noindent
  \optarg[#2]{#2}%
  \normalsize
  \hfill
  \optarg[#3]{#3}%
}

}
